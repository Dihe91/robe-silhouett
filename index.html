const ScrollProgress = () => {

  const [width, setWidth] = useState(0);

  useEffect(() => {

    const handleScroll = () => {

      const scrollTotal = document.documentElement.scrollHeight - document.documentElement.clientHeight;

      setWidth((window.scrollY / scrollTotal) * 100);

    };

    window.addEventListener('scroll', handleScroll);

    return () => window.removeEventListener('scroll', handleScroll);

  }, []);

  return <div className="fixed top-0 left-0 h-[2px] bg-black z-[200] transition-all duration-100" style={{ width: `${width}%` }} />;

};


const LiveNotification = ({ t }) => {

  const [visible, setVisible] = useState(false);

  const [current, setCurrent] = useState(0);

  

  // Liste élargie de clients pour éviter les répétitions rapides

  const users = [

    "Léa (Paris)", "Sarah (Lyon)", "Amira (Dubai)", "Elena (Madrid)", 

    "Chloé (Bordeaux)", "Inès (Marseille)", "Sophie (Bruxelles)", 

    "Camille (Genève)", "Lina (Casablanca)", "Emma (Londres)",

    "Clara (Nantes)", "Manon (Lille)", "Jade (Toulouse)", "Zoé (Nice)"

  ];

  

  const usedIndices = useRef([]);


  useEffect(() => {

    let timeoutId;


    const showNextNotification = () => {

      let availableIndices = users

        .map((_, i) => i)

        .filter(i => !usedIndices.current.includes(i));


      // Si tous les prénoms ont été utilisés, on réinitialise la pioche

      if (availableIndices.length === 0) {

        usedIndices.current = [];

        availableIndices = users.map((_, i) => i);

      }


      // Tire un prénom au hasard parmi ceux restants

      const randomIndex = availableIndices[Math.floor(Math.random() * availableIndices.length)];

      usedIndices.current.push(randomIndex);

      

      setCurrent(randomIndex);

      setVisible(true);


      // Cache la notification après 5 secondes

      setTimeout(() => setVisible(false), 5000);


      // Programme la prochaine entre 1 minute (60 000ms) et 4 minutes (240 000ms)

      const nextInterval = Math.floor(Math.random() * (240000 - 60000 + 1)) + 60000;

      timeoutId = setTimeout(showNextNotification, nextInterval);

    };


    // Première notification après un court délai (entre 15s et 45s)

    const initialDelay = Math.floor(Math.random() * (45000 - 15000 + 1)) + 15000;

    timeoutId = setTimeout(showNextNotification, initialDelay);


    return () => clearTimeout(timeoutId);

  }, []);


  if (!visible) return null;


  return (

    <div className="fixed bottom-6 left-6 z-[150] bg-white border border-gray-100 p-4 flex items-center gap-4 shadow-xl animate-fade-in-up">

      <div className="w-8 h-8 bg-gray-900 flex items-center justify-center rounded-full">

        <Bell size={14} className="text-white" />

      </div>

      <div className="text-[10px] uppercase tracking-widest leading-none text-black">

        <p className="font-bold">{users[current]}</p>

        <p className="text-gray-400 mt-1">{t.liveBuy} <strong>ROBE MINUIT</strong></p>

      </div>

    </div>

  );

};

                  {selectedSize && stockVal <= 3 && stockVal > 0 && <p className="text-[10px] text-orange-500 font-bold uppercase tracking-widest animate-pulse">{stockVal === 1 ? t.lastOne : t.onlyLeft.replace('{n}', stockVal)}</p>}

                  {selectedSize && stockVal === 0 && <p className="text-[10px] text-red-500 font-bold uppercase tracking-widest">{t.outOfStock}</p>}

              </div>

            </div>


            <div className="space-y-4">

              <button disabled={!selectedSize || stockVal === 0} onClick={() => handleAction(false)} 

                className={`w-full py-5 text-[10px] font-bold uppercase tracking-widest transition-all flex items-center justify-center gap-2 ${added ? 'bg-green-600 text-white border-green-600' : 'bg-white text-black border border-black hover:bg-black hover:text-white disabled:opacity-20'}`}

              >

                {added ? <><Check size={14}/> {t.added}</> : t.addToCart}

              </button>

              <button disabled={!selectedSize || stockVal === 0} onClick={() => handleAction(true)} 

                className="w-full py-5 bg-black text-white text-[10px] font-bold uppercase tracking-widest hover:opacity-80 transition disabled:opacity-20"

              >

                {t.buyNow}

              </button>

            </div>


            <div className="mt-16 pt-8 border-t border-gray-50">

              <h3 className="text-[10px] font-bold uppercase mb-4 text-gray-400">{t.description}</h3>

              <p className="text-xs leading-relaxed text-gray-600 font-light uppercase tracking-tight">{PRODUCT_INFO.description}</p>

            </div>

          </div>

        </div>

      </section>

    </div>

  );

};

// --- 8. APPLICATION PRINCIPALE ---

export default function App() {

  const [view, setView] = useState('home');

  const [cart, setCart] = useState([]);

  const [stock, setStock] = useState(Object.keys(VARIANTS).reduce((acc, k) => ({...acc, [k]: {...VARIANTS[k].stock}}), {}));

  const [lang, setLang] = useState('fr');

  const [isMenuOpen, setIsMenuOpen] = useState(false);


  useEffect(() => {

    if (!document.querySelector('#tailwind-cdn')) {

      const s = document.createElement('script');

      s.id = 'tailwind-cdn'; s.src = "https://cdn.tailwindcss.com";

      document.head.appendChild(s);

    }

  }, []);


  const t = TRANSLATIONS[lang] || TRANSLATIONS['fr'];

  const isRTL = lang === 'ar';


  const handleCheckout = () => {

    const newStock = { ...stock };

    cart.forEach(item => { if (newStock[item.color][item.size] > 0) newStock[item.color][item.size] -= 1; });

    setStock(newStock);

    setCart([]);

    alert(lang === 'ar' ? "تم تأكيد طلبك!" : "Commande confirmée !");

    setView('home');

  };


  return (

    <div className={`font-sans antialiased text-black bg-white selection:bg-black selection:text-white ${isRTL ? 'font-arabic' : ''}`} dir={isRTL ? 'rtl' : 'ltr'}>

      <style>{`

        @import url('https://fonts.googleapis.com/css2?family=Noto+Sans+Arabic:wght@400;700&family=Playfair+Display:wght@700&display=swap');

        .font-arabic { font-family: 'Noto Sans Arabic', sans-serif; }

        .font-serif { font-family: 'Playfair Display', serif; }

        .scrollbar-hide::-webkit-scrollbar { display: none; }

        .scrollbar-hide { -ms-overflow-style: none; scrollbar-width: none; }

        @keyframes fade-in { from { opacity: 0; } to { opacity: 1; } }

        .animate-fade-in { animation: fade-in 1s ease-out; }

        @keyframes slide-up { from { opacity: 0; transform: translateY(30px); } to { opacity: 1; transform: translateY(0); } }

        .animate-slide-up { animation: slide-up 1s cubic-bezier(0.16, 1, 0.3, 1) forwards; }

      `}</style>


      <ScrollProgress />

      <Header cartCount={cart.length} setView={setView} lang={lang} setLang={setLang} t={t} onMenuOpen={() => setIsMenuOpen(true)} />

      <Sidebar isOpen={isMenuOpen} onClose={() => setIsMenuOpen(false)} t={t} setView={setView} isRTL={isRTL} />

      <LiveNotification t={t} />


      <main>

        {view === 'home' && <ShowcasePage addToCart={(p) => setCart([...cart, p])} setView={setView} t={t} currentStock={stock} isRTL={isRTL} />}

        {view === 'cart' && <CartPage cart={cart} removeFromCart={(id) => setCart(cart.filter(i => i.cartId !== id))} setView={setView} t={t} onCheckout={handleCheckout} />}

        {view === 'about' && <AboutPage t={t} />}

        {view === 'legal' && <LegalPage t={t} />}

      </main>


      <footer className="py-20 border-t border-gray-50 text-center bg-white">

        <h2 className="text-4xl font-serif font-bold uppercase tracking-tighter mb-8">Silhouett.</h2>

        <div className="flex justify-center space-x-10 text-[10px] font-bold uppercase text-gray-400 tracking-widest rtl:space-x-reverse">

           <a href="#" className="hover:text-black transition-colors">Instagram</a>

           <a href="#" className="hover:text-black transition-colors">Contact</a>

           <button onClick={() => { setView('legal'); window.scrollTo(0,0); }} className="hover:text-black transition-colors uppercase">Légal</button>

        </div>

        <p className="mt-12 text-[9px] text-gray-300 uppercase tracking-widest">Studio Paris • 2026</p>

      </footer>

    </div>

  );

}
